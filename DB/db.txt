1. DB ver1

테이블 이름 : user => 아이디(key) 비밀번호 닉네임 이름 
create table user (id varchar(20) not null,
   password varchar(20) not null,
   nickname varchar(20) not null,
   name varchar(20) not null,
   primary key(id));


테이블 이름 : post -> 게시글 index(int,key), 아이디(id) , 파일(이미지. 동영상)(varchar(20), 내용(txt), 댓글(id), 좋아요(id1 id2 ---), 시간(date)
create table post (post_id int(20) not null auto_increment,
   id varchar(20) not null,
   file varchar(50) not null,
   content text,
   comment varchar(20),
   likes varchar(20),
   upload_date datetime not null,
   primary key(post_id));


테이블 이름 : user_post -> 아이디, 게시글(index), 팔로우(id), 팔로워(id), 북마크한 게시글(id),
create table user_post(id varchar(20) not null, post_index varchar(100), follow text, follower text, bookmark_post varchar(100));


테이블 이름 : post_comment -> 아이디(id) , 게시글 index(int), 내용(txt), 좋아요(id), 시간(date)
create table post_comment (id varchar(20) not null,
   post_index int(20) not null,
   content text not null,
   likes varchar(20),
   comment_date datetime not null);


2. ver2----------------

테이블 이름 : user => 아이디(key) 비밀번호 닉네임 이름 
create table user (id varchar(20) not null,
   password varchar(20) not null,
   nickname varchar(20) not null,
   name varchar(20) not null,
   primary key(id));

테이블 이름 : post -> 게시글 index(int,key), 아이디(id) , 시간(date)
create table post (post_id int(20) not null ,
   id varchar(20) not null,
   nickname varchar(20) not null,
   upload_date datetime not null default current_timestamp,
   primary key(post_id));

테이블 이름: post_image => 게시글 index(int), image(varchar)
create table post_image (post_id int(20) not null,
   image_link varchar(20) not null);

테이블 이름: post_content => 게시글 index(int), content(varchar)
create table post_content (post_id int(20) not null,
   content text not null);

테이블 이름: post_comment => 게시글 index(int), image(varchar)
create table post_comment (post_id int(20) not null,
   comment_comtent varchar(128) not null,
   upload_date datetime not null);

테이블 이름: post_likes => 게시글 index(int), image(varchar)
create table post_likes (post_id int(20) not null,
   likes_id varchar(20) not null);

테이블 이름: following => id, follw id
create table following (id varchar(20) not null,
   following_id varchar(20) not null);

테이블 이름: follower => id, follw id
create table follower(id varchar(20) not null,
